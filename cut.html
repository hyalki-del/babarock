// Replace with your actual Spreadsheet ID const SPREADSHEET_ID = "1a2b3c4d5e6f7g8h9_YOUR_SHEET_ID"; const SHEET_NAME = "keys"; // Tab name in your Google Sheet // Direct Google Sheet export endpoint (No "Publish to Web" required) const SHEET_CSV_URL = `https://docs.google.com/spreadsheets/d/${SPREADSHEET_ID}/gviz/tq?tqx=out:csv&sheet=${encodeURIComponent(SHEET_NAME)}`; function parseCSV(text) { const lines = text.trim().split(/\r?\n/); if (lines.length < 2) return []; // Parse header row const headers = lines[0].split(',').map(h => h.trim().toLowerCase().replace(/^"|"$/g, '')); const data = []; for (let i = 1; i < lines.length; i++) { // Handle values wrapped in quotes const row = lines[i].match(/(".*?"|[^",\s]+)(?=\s*,|\s*$)/g) || lines[i].split(','); const cleanRow = row.map(val => val.trim().replace(/^"|"$/g, '')); const obj = {}; headers.forEach((header, index) => { obj[header] = cleanRow[index] || ""; }); data.push(obj); } return data; } async function loadPlaylist() { const listElement = document.getElementById("songList"); const countElement = document.getElementById("songCount"); try { // Bust browser cache with a dynamic timestamp const res = await fetch(`${SHEET_CSV_URL}&t=${Date.now()}`); if (!res.ok) throw new Error(`HTTP error! status: ${res.status}`); const csvText = await res.text(); const playlist = parseCSV(csvText); listElement.innerHTML = ""; countElement.textContent = `${playlist.length} Songs`; playlist.forEach((item, index) => { const li = document.createElement("li"); const query = encodeURIComponent(`${item.title} ${item.artist}`); const youtubeUrl = `https://www.youtube.com/results?search_query=${query}`; const keyBadgeHtml = (item.key && item.key.toUpperCase() !== 'N/A') ? `${item.key}` : ''; li.innerHTML = `
${index + 1}
${item.title} ${item.artist}
${keyBadgeHtml}
`; listElement.appendChild(li); }); } catch (err) { console.error("Error fetching live sheet data:", err); countElement.textContent = "Error"; } } loadPlaylist();
